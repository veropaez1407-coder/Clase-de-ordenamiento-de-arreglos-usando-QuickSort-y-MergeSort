public class Arreglo {
    //atributosss
    private int[] numeros;
    private int tamaño;
    //constructor
    public Arreglo(int[] numeros) {
        this.numeros = numeros;
        this.tamaño = numeros.length;
    }
    //metodos
    public int[] getNumeros() {
        return numeros;
    }
    public void Show(String mensaje) {
        System.out.println(mensaje + ": ");
        for (int i = 0; i < tamaño; i++) {
            System.out.print(numeros[i] + " ");
        }
        System.out.println();
    }
    public int[] hacerCopiaaaa() {
        int[] copia = new int[tamaño];
        for (int i = 0; i < tamaño; i++) {
            copia[i] = numeros[i];
        }
        return copia;
    }
}
